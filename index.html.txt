  <script>
    // Mock data (después lo reemplazamos con API real)
    let currentTimeframe = 1;
    let items = [
      {name:"Revolution Case", price:0.92, change24:22.7, change7d:41.3, change30d:68.9, volume:12450, stock:8920},
      {name:"Kilowatt Case", price:2.18, change24:15.6, change7d:-8.3, change30d:42.1, volume:9800, stock:6730},
      {name:"AK-47 | Redline (FT)", price:12.45, change24:8.3, change7d:-4.2, change30d:15.6, volume:1240, stock:856},
      {name:"Paris 2023 Legends Capsule", price:4.20, change24:-12.5, change7d:18.9, change30d:-22.4, volume:3240, stock:2890},
      {name:"AWP | Dragon Lore (FN)", price:12450, change24:-2.8, change7d:5.1, change30d:12.3, volume:67, stock:12},
    ];

    function getChange(item) {
      if (currentTimeframe === 1) return item.change24;
      if (currentTimeframe === 2) return item.change7d;
      return item.change30d;
    }

    function renderTopCards() {
      const gainersDiv = document.getElementById('gainers');
      const losersDiv = document.getElementById('losers');
      gainersDiv.innerHTML = losersDiv.innerHTML = '';

      const sortedGainers = [...items].sort((a,b) => getChange(b) - getChange(a)).slice(0,4);
      const sortedLosers = [...items].sort((a,b) => getChange(a) - getChange(b)).slice(0,4);

      sortedGainers.forEach(item => {
        const change = getChange(item);
        gainersDiv.innerHTML += `
          <div class="bg-zinc-800 p-5 rounded-2xl card cursor-pointer hover:border-green-500/50" onclick="loadChart('${item.name}')">
            <div class="text-zinc-400 text-sm truncate">${item.name}</div>
            <div class="flex justify-between items-end mt-3">
              <span class="text-2xl font-mono font-bold">$${item.price.toFixed(2)}</span>
              <span class="text-emerald-400 text-2xl font-bold">+${change}%</span>
            </div>
          </div>`;
      });

      sortedLosers.forEach(item => {
        const change = getChange(item);
        losersDiv.innerHTML += `
          <div class="bg-zinc-800 p-5 rounded-2xl card cursor-pointer hover:border-red-500/50" onclick="loadChart('${item.name}')">
            <div class="text-zinc-400 text-sm truncate">${item.name}</div>
            <div class="flex justify-between items-end mt-3">
              <span class="text-2xl font-mono font-bold">$${item.price.toFixed(2)}</span>
              <span class="text-red-400 text-2xl font-bold">${change}%</span>
            </div>
          </div>`;
      });
    }

    function changeTimeframe(tf) {
      currentTimeframe = tf;
      document.querySelectorAll('.time-btn').forEach(btn => {
        btn.classList.remove('bg-green-500', 'text-black', 'active');
      });
      event.target.classList.add('bg-green-500', 'text-black', 'active');
      document.querySelector('.change-header').textContent = ['','24H','7D','30D'][tf];
      renderTopCards();
    }

    function loadChart(name) {
      document.getElementById('chart-title').textContent = name;
      // Gráfico básico mock (después lo hacemos real con datos históricos)
      const ctx = document.getElementById('priceChart').getContext('2d');
      if (window.myChart) window.myChart.destroy();
      window.myChart = new Chart(ctx, {
        type: 'line',
        data: {
          labels: ['Hace 30d', 'Hace 15d', 'Hace 7d', 'Ayer', 'Hoy'],
          datasets: [{
            label: 'Precio',
            data: [100, 120, 110, 130, 145],
            borderColor: '#22c55e',
            tension: 0.4
          }]
        },
        options: { plugins: { legend: { display: false } } }
      });
    }

    // Inicializar
    renderTopCards();
    loadChart(items[0].name);
  </script>
